$vSwitchName = "Mirror Traffic"

# Get the Ethernet Switch Port Security Settings feature
$feature = Get-VMSystemSwitchExtensionPortFeature -FeatureName "Ethernet Switch Port Security Settings"

# Set MonitorMode to 2 (Source) to enable port mirroring
$feature.SettingData.MonitorMode = 2

# Add the feature to the external port of the virtual switch
Add-VMSwitchExtensionPortFeature -ExternalPort -SwitchName $vSwitchName -VMSwitchExtensionFeature $feature
